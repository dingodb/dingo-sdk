# ThreadSanitizer suppressions for sdk_unit_test.
#
# The dingo-eureka third-party libraries (protobuf, brpc, ...) are prebuilt
# without TSan instrumentation, so their internal synchronization is invisible
# to TSan. Keep entries narrow: suppress only known-safe internal lazy-init
# paths, never whole libraries, so real races on message objects still show.

# protobuf descriptor lazy initialization is guarded by an internal call_once
# TSan cannot see.
race:google::protobuf::*OnceInit*
race:google::protobuf::DescriptorPool::Tables::AllocateBytes

# Pre-existing Timer reports in the std::thread actuator (double lock in
# Timer::Add vs Run/Stop plus races on FunctionInfo). They reproduce on a
# pristine tree and are tracked separately; suppressed here so they do not
# mask task-level reports. Remove once the Timer is fixed.
mutex:dingodb::sdk::Timer::*
race:dingodb::sdk::Timer::*
